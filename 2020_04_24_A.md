###### 이미지 1초마다 잘라서 저장하기

```python
from moviepy.editor import *

videoPath= 'examples/project_video.mp4'
my_clip = VideoFileClip(videoPath)
duration = int(my_clip.duration)

for i in range(0, duration) :
    my_clip.save_frame('test_pic/pic' + str(i+1) + '.jpg', i)
```
